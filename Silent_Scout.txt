// NOTHING TO EDIT BELOW HERE

echo "Beginning scout all cities script..."
execute "scanmap {city.coords} {radius}"
allCities = FindField(city.x,city.y,radius,11).sort(city.compareByDistanceToCastle)
echo "Found {allCities.length} cities within {radius} miles of this city."
label checkCity
if allCities.length = 0 goto allCitiesScouted
thisCity = allCities.shift()
// get details on coords, but if server is laggy skip coords
attempts = 0
label getDetails
details = GetDetailInfo(thisCity, false, 30) 
if details goto gotDetails
if ++attempts < 10 goto getDetails
echo "Server is being laggy, cannot get details on {FieldIdToCoords(thisCity)}, skipping."
goto checkCity
label gotDetails
// if you can't scout it, or its in a neutral alliance skip to next city
if ((!details.canScout) || (RelationIndex(details) == 4)) goto checkCity
// wait for open rally spot (keep 1 open for trainer/ hiding)
label waitForRally
if city.rallySpotAvailable(1,1) goto scoutIt
echo "Waiting for free rally spot, sleeping for 1 minute..."
sleep 60
goto waitForRally
label scoutIt
echo "Scouting {details.name}({FieldIdToCoords(thisCity)}) {details.userName} [{details.allianceName}]"
execute "scout {FieldIdToCoords(thisCity)} none {scoutWave}"
goto checkCity
// done scouting
label allCitiesScouted
config wartown:0
echo "Finished scouting! You should review & process reports now."
// sleep for an hour
sleep 60:00
goto autorun
